// Project Structure Documentation
// ================================

// 📁 src/
// ├── assets/ # Images, icons, static files
// ├── components/ # Reusable React components
// │ ├── common/ # Shared by both user and admin
// │ ├── admin/ # Admin-specific components
// │ └── user/ # User-specific components
// ├── layouts/ # Layout wrapper components
// │ ├── UserLayout.jsx
// │ └── AdminLayout.jsx
// ├── pages/ # Full page components
// │ ├── admin/ # Admin pages
// │ └── user/ # User pages
// ├── routes/ # React Router configuration
// │ └── index.jsx # Main router setup
// ├── services/ # API calls and external services
// │ ├── userService.js # User API calls
// │ └── adminService.js # Admin API calls
// ├── store/ # Global state management (Redux, Zustand, etc.)
// ├── utils/ # Helper functions
// │ └── helpers.js # Common utility functions
// ├── App.jsx # App root component
// └── main.jsx # Entry point

// 🚀 Getting Started:
// 1. Run: npm install
// 2. Run: npm run dev
// 3. Modify pages and components as needed
// 4. Add your API endpoints in services/
// 5. Use routes/index.jsx to add new routes
